
# Helm Service: Create NS and install chart!
kubectl create ns delivery-simplenode-prod
helm upgrade --install simplenode . -n delivery-simplenode-prod

# Rollout: Validate that rollouts are listed!
kubectl argo rollouts list rollouts -n delivery-simplenode-prod

# Rollout: Watch rollout status
kubectl argo rollouts get rollout simplenode-prod -n delivery-simplenode-prod -w


kubectl argo rollouts promote simplenode-prod -n delivery-simplenode-prod
